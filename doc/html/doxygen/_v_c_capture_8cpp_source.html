94 bool CVCCapture::CaptureImage(CByteImage ** ppImages)
95 {
96  //TODO: allow background image capture
97  if (tpp() == -1)
98  return false;
99 
100  //get the capture buffer address here and not in constructor as someone might have changed the location
101  unsigned char* restrict scr_pixels = (unsigned char*) ScrGetCaptPage;
102  unsigned char* restrict img_pixels = ppImages[0]->pixels;
103 
104  int h = std::min(m_height, ppImages[0]->height);
105  int w = std::min(m_width, ppImages[0]->width);
106 
107  //copy row by row as width may differ and pitch most likely will
108  for (int row=0; row<h; row++) {
109  std::copy(scr_pixels, scr_pixels+w, img_pixels);
110 
111  scr_pixels += m_pitch;
112  img_pixels += ppImages[0]->width;
113  }
114 
115  return true;
116 }
