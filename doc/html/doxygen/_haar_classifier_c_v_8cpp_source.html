90 int CHaarClassifierCV::Find(CByteImage *pImage, RegionList &regionList)
91 {
92  if (!m_pCascade)
93  return -1;
94 
95  regionList.clear();
96 
97  cvClearMemStorage(m_pStorage);
98 
99  IplImage *pIplImage = IplImageAdaptor::Adapt(pImage);
100 
101  const CvSeq *pFaces = cvHaarDetectObjects(pIplImage, m_pCascade, m_pStorage, 1.3, 3, CV_HAAR_DO_CANNY_PRUNING, cvSize(40, 40));
102  const int nFaces = pFaces ? pFaces->total : 0;
103 
104  for (int i = 0; i < nFaces; i++)
105  {
106  CvRect* r = (CvRect *) cvGetSeqElem(pFaces, i);
107 
108  MyRegion region;
109  region.min_x = r->x;
110  region.min_y = r->y;
111  region.max_x = r->x + r->width;
112  region.max_y = r->y + r->height;
113 
114  regionList.push_back(region);
115  }
116 
117  cvReleaseImageHeader(&pIplImage);
118 
119  return nFaces;
120 }
