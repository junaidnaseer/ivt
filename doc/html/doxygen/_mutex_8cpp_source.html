95 // try to lock the mutex. If another thread is owner of mutex, return EBusy, otherwise lock mutex
96 Threading::EMutexStatus CMutex::TryLock()
97 {
98 #ifdef WIN32
99  if (TryEnterCriticalSection(&m_CSWindowsMutex) == 0)
100  {
101  return Threading::eMutexBusy;
102  }
103  else
104  {
105  return Threading::eMutexSuccess;
106  }
107 #else
108  const int nResult = pthread_mutex_trylock(&m_PosixMutex);
109 
110  if (nResult == 0)
111  {
112  return Threading::eMutexSuccess;
113  }
114  else if (nResult == EBUSY)
115  {
116  return Threading::eMutexBusy;
117  }
118  else
119  {
120  return Threading::eMutexError;
121  }
122 #endif
123 }
