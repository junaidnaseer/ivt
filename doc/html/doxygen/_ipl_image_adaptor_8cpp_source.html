66 
67  IplImage *pIplImage;
68 
69  if (bAllocateMemory)
70  {
71  pIplImage = cvCreateImage(cvSize(pImage->width, pImage->height), IPL_DEPTH_8U, pImage->bytesPerPixel);
72  memcpy(pIplImage->imageData, pImage->pixels, pImage->width * pImage->height * pImage->bytesPerPixel);
