102 int CUndistortion::Init(const char *pCameraParameterFileName)
103 {
104  if (m_pStereoCalibration->LoadCameraParameters(pCameraParameterFileName))
105  {
106  // stereo calibration
107  m_pCalibrationLeft = m_pStereoCalibration->GetLeftCalibration();
108  m_pCalibrationRight = m_pStereoCalibration->GetRightCalibration();
109 
110  UpdateMaps();
111 
112  return 2;
113  }
114  else
115  {
116  // not a stereo calibration
117  if (m_pCalibration->LoadCameraParameters(pCameraParameterFileName))
118  {
119  m_pCalibrationLeft = m_pCalibration;
120  m_pCalibrationRight = 0;
121 
122  UpdateMaps();
123 
124  return 1;
125  }
126  }
127 
128  return 0;
129 }
